280 /**
281  A class managing basic ref counting references retain/release mechansim for use in
282  instance variables which will not allow constructors or destructors. Use the &operator
283  to get a pointer with "autorelease" scope for use in the rest of your program.
284 
285  To free the ref either assign nil references the "=" operator or call use the ~ operator
286  which returns a transient ref. Use &~var to free the ref with autorelrease scope.
287 
288  Usage:
289  <pre>
290  OOReference<NSString *> immutableRef = @"STRING"; // take referrence to a string
291  OOReference<NSMutableArray *> mutableRef <<= [NSArray array]; // take mutable copy
292  NSMutableArray *ptr = &amp;mutableRef; // get autoreleasing pointer to original object
293  NSMutableArray *ptr = *mutableRef; // get pointer to original object
294  ~mutableRef; // clear out pointer and remove "release" object.
295 
296  - (NSString *)function:(NSString *)str {
297  OOReference<NSString *> ref = str; // take ref
298 
299  // do something
300 
301  return &amp;str; // get autoreleasing pointer to return
302  // ref is discarded when object destructed on function exit.
303  }
304  </pre>
305  */

312  oo_inline Class classOfReference() const {

319  ref = nil;

323 protected:

331 #ifdef __OBJC_GC__
332  oo_inline virtual id rawset( RTYPE val ) OO_RETURNS {
333 #else
334  oo_inline RTYPE rawset( RTYPE val ) OO_RETURNS {
335 #endif
336  RTYPE old = ref;
337  ref = val;
338  if ( val != nil && val != (id)kCFNull ) {
339  OORetain( @"%p %s#%ld: %p = %@", this, "RETAIN",

341  }
342  if ( old != nil && old != (id)kCFNull ) {
343  OORelease( @"%p %s#%ld: %p = %@", this, "RELEASE",
344  (OOLong)OO_RETAINCOUNT( old ), old, old );
345  OO_RELEASE( old );
346  old = nil;
347  }
348  return ref;
349  }

353  return rawset( OO_RETAIN( val ) );
