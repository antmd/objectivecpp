167 /**
168  Simple persistence class for Objective-C objects using a "sqlite" database. Instances a
169  class are automatically bound to a table with the class name for insert, update and delete.
170 
171  A table with the same name as the class is created automatically when you perform any
172  operation on instances of that class with columns corresponding to the instance variables
173  of the class except those which begin with the character "_". If an instance variable
174  starts with an upper case letter it will be indexed.
175 
176  To update a record, fetch it and call the update: method passing it in as the argument to
177  save it's preious values. You can then modify it and call commit to update it's record.
178  Delete operations on objects also accumulate and need to be commited to take effect.
179  */

283  oo_inline OOOODatabase() {
284  autocommit = YES;
285  autorelease = NO;
286  }
287  oo_inline int commit() {
288  return [OODatabase commit];
289  }
290  oo_inline id autowhatever( id record ) {
291  if ( autocommit )
292  commit();
293  if ( autorelease )
294  OO_RELEASE( record );
295  return autorelease ? nil : record;
296  }
297  oo_inline id operator += ( id record ) {
298  [[OODatabase sharedInstance] insert:record];
299  return autowhatever( record );
300  }
301  oo_inline id operator -= ( id record ) {
302  [[OODatabase sharedInstance] delete:record];
303  return autowhatever( record );
304  }
305  oo_inline id operator *= ( id record ) {
306  [[OODatabase sharedInstance] update:record];
307  return record;
308  }
