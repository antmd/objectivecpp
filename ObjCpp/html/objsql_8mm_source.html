267 - (OOStringArray)registerSubclassesOf:(Class)recordSuperClass {
268  int numClasses = objc_getClassList( NULL, 0 );
269  Class *classes = (Class *)malloc( sizeof *classes * numClasses );
270  OOArray<Class> viewClasses;
271  OOStringArray classNames;
272 
273  // scan all registered classes for relevant subclasses
274  numClasses = objc_getClassList( classes, numClasses );
275  for ( int c=0 ; c<numClasses ; c++ ) {
276  Class superClass = classes[c];
277  if ( class_getName( superClass )[0] != '_' )
278  while ( (superClass = class_getSuperclass( superClass )) )
279  if ( superClass == recordSuperClass ) {
280  if ( [classes[c] respondsToSelector:@selector(ooTableSql)] )
281  viewClasses += classes[c];
282  else {
283  [[OODatabase sharedInstance] tableMetaDataForClass:classes[c]];
284  classNames += class_getName( classes[c] );
285  }
286  break;
287  }
288  }
289 
290  // delay creation views until after tables
291  for ( int c=0 ; c<viewClasses ; c++ ) {
292  [[OODatabase sharedInstance] tableMetaDataForClass:viewClasses[c]];
293  classNames += class_getName( viewClasses[c] );
294  }
295 
296  // return classes in order registered
297  free( classes );
298  return classNames;
299 }

362 
363  id value = values[name];
364  if ( !value || value == OONull )
365  out += qualifyNulls ? OOFormat( @"%s\n\t%@ is NULL", prefix, name ) : @"";
366  else if ( !qualifyNulls && [value isKindOfClass:[NSString class]] &&
367  [value rangeOfString:@"%"].location != NSNotFound )
368  out += OOFormat( @"%s\n\t%@ LIKE ?", prefix, name );
369  else
370  out += OOFormat( @"%s\n\t%@ = ?", prefix, name );
371  }

574 /**
575  Commit all pending inserts, updates and deletes to the database. Use commitTransaction to perform
576  this inside a database transaction.
577  */
578 
579 - (int)commit {
580  int commited = 0;
581 
582  for ( int i=0 ; i<transaction ; i++ ) {
583  OOValueDictionary values = transaction[i];
584  OOString exec = (NSMutableString *)~values[kOOExecSQL];
585  if ( !!exec ) {
586  if ( ![self exec:@"%@", *exec] )
587  OOWarn( @"-[ODatabase commit] Error in transaction exec: %@ - %s", *exec, errmsg );
588  continue;
589  }
590 
591  OORef<NSObject *> object = *values[kOOObject]; values -= kOOObject;
592  BOOL isInsert = !!~values[kOOInsert], isUpdate = !!~values[kOOUpdate];
593 
594  OOMetaData *metaData = [self tableMetaDataForClass:[object class]];
595  OOValueDictionary newValues = [metaData encode:[object dictionaryWithValuesForKeys:metaData->columns]];
596  OOStringArray changedCols;
597 
598  if ( isUpdate ) {
599  for ( NSString *name in *metaData->columns )
600  if ( ![*newValues[name] isEqual:values[name]] )
601  changedCols += name;
602  }
603  else
604  values = newValues;
605 
606  OOString sql = isInsert ?
607  OOFormat( @"insert into %@ (%@) values (", *metaData->tableName, *(metaData->columns/", ") ) :
608  OOFormat( isUpdate ? @"update %@ set" : @"delete from %@", *metaData->tableName );
609 
610  int nchanged = changedCols;
611  if ( isUpdate && nchanged == 0 ) {
612  OOWarn( @"%s %@ (%@)", errmsg = (char *)"-[ODatabase commit:] Update of unchanged record", *object, *(lastSQL = sql) );
613  continue;
614  }
615 
616  for ( int i=0 ; i<nchanged ; i++ )
617  sql += OOFormat( @"%s\n\t%@ = ?", i==0 ? "" : ",", **changedCols[i] );
618 
619  if ( isInsert ) {
620  OOString quote = "?", commaQuote = ", ?";
621  for ( int i=0 ; i<metaData->columns ; i++ )
622  sql += i==0 ? quote : commaQuote;
623  sql += ")";
624  }
625  else
626  sql += [self whereClauseFor:metaData->columns values:values qualifyNulls:YES];
627 
628 #ifdef OODEBUG_SQL
629  NSLog( @"-[OODatabase commit]: %@ %@", *sql, *values );
630 #endif
631 
632  if ( ![*adaptor prepare:sql] )
633  continue;
634 
635  if ( isUpdate )
636  [*adaptor bindCols:changedCols values:newValues startingAt:1 bindNulls:YES];
637  [*adaptor bindCols:metaData->columns values:values startingAt:1+nchanged bindNulls:isInsert];
638 
639  [*adaptor bindResultsIntoInstancesOfClass:nil metaData:metaData];
640  commited += updateCount;
641  }
642 
643  transaction = nil;
644  return commited;
645 }

801 /**
802  Return a dictionary containing the values for a row returned by the database from a select.
803  These values need to be decoded using a classes metadata to set the ivar values later.
804  */
805 
806 - (OOValueDictionary)valuesForNextRow {
807  int ncols = sqlite3_column_count( stmt );
808  OOValueDictionary values;
809 
810  for ( int i=0 ; i<ncols ; i++ ) {
811  OOString name = sqlite3_column_name( stmt, i );
812  id value = nil;
813 
814  switch ( sqlite3_column_type( stmt, i ) ) {
815  case SQLITE_NULL:
816  value = OONull;
817  break;
818  case SQLITE_INTEGER:
819  value = [[NSNumber alloc] initWithLongLong:sqlite3_column_int64( stmt, i )];
820  break;
821  case SQLITE_FLOAT:
822  value = [[NSNumber alloc] initWithDouble:sqlite3_column_double( stmt, i )];
823  break;
824  case SQLITE_TEXT: {
825  const unsigned char *bytes = sqlite3_column_text( stmt, i );
826  value = [[NSMutableString alloc] initWithBytes:bytes
827  length:sqlite3_column_bytes( stmt, i)
828  encoding:NSUTF8StringEncoding];
829  }
830  break;
831  case SQLITE_BLOB: {
832  const void *bytes = sqlite3_column_blob( stmt, i );
833  value = [[NSData alloc] initWithBytes:bytes length:sqlite3_column_bytes( stmt, i )];
834  }
835  break;
836  default:
837  OOWarn( @"-[OOAdaptor valuesForNextRow:] Invalid type on bind of ivar %@: %d", *name, sqlite3_column_type( stmt, i ) );
838  }
839 
840  values[name] = value;
841  OO_RELEASE( value );
842  }
843 
844  return values;
845 }
